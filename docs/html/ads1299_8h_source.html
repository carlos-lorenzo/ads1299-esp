141/*
142 * Circular buffer of fully parsed ADS1299 chunks.
143 *
144 * Layout of buf[]:
145 *
146 * slot 0 slot 1
147 * [s0|s1|...|sN-1] [s0|s1|...|sN-1] ...
148 * ^ ^
149 * buf[0] buf[chunk_samples]
150 *
151 * A slot is addressed as: buf[(index & mask) * chunk_samples + sample_i]
152 *
153 * capacity slots are physically allocated.
154 * capacity-1 slots are usable (one sentinel slot disambiguates full from empty).
155 *
156 * SPSC ownership:
157 * head — written exclusively by ISR (post_cb)
158 * tail — written exclusively by handler task
159 */
160typedef struct {
161 ads1299_sample_t *buf; /* flat array: capacity * chunk_samples entries */
162 uint32_t capacity; /* total slot count; must be power-of-2, >= 2 */
163 uint32_t chunk_samples; /* samples per slot */
164 uint32_t mask; /* capacity - 1; used for slot index modulo */
165 volatile uint32_t head; /* next write slot index; ISR-owned */
166 volatile uint32_t tail; /* next read slot index; task-owned */
167} ads1299_chunkring_t;
